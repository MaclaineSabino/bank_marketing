from sklearn.metrics import confusion_matrix
def plot_confusion_matrix(cm, title='Matriz da confuso', cmap=plt.cm.Blues):
    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title)
    plt.colorbar()    
    plt.tight_layout()
    plt.ylabel('Real')
    plt.xlabel('Predito')
    plt.show

from sklearn.metrics import confusion_matrix

def fit_and_predict(nome,modelo,treino_dados,treino_marcacoes,test_dados,test_marcacoes):
    modelo.fit(treino_dados,treino_marcacoes)
    resultado = modelo.predict(test_dados)
    acertos = 0
    tamanho = len(test_marcacoes)
    for i in range(tamanho):
        if test_marcacoes[i]==resultado[i]:
            acertos = acertos+1
    perc = acertos*100/tamanho
    print('Acerto %s: %.2f%%'%(nome,perc))
    
    
    return perc
